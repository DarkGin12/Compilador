    Example 2: fibonacci function

    var fibonacci = function(n) {
    var a = 0, b = 1;
    for (var i = 2; i <= n; i++) {
        var f = a;
        a = b;
        b += f;
    }
    return b;
};

console.log(fibonacci(14)) Saida : 377

Exemplo de uso da bibloteca Axios

  
    

// Função para fazer uma requisição GET
function fetchData() {
    axios.get('https://jsonplaceholder.typicode.com/posts/1')
        .then(response => {
            console.log('GET Response:', response.data);
        })
        .catch(error => {
            console.error('Error fetching data:', error);
        });
}

    // Função para fazer uma requisição POST
    function postData() {
        const data = {
            title: 'foo',
            body: 'bar',
            userId: 1,
    };

    axios.post('https://jsonplaceholder.typicode.com/posts', data)
        .then(response => {
            console.log('POST Response:', response.data);
        })
        .catch(error => {
            console.error('Error posting data:', error);
        });
}

    // Chamar as funções
    fetchData();
    postData();

Exemplo de Uso da Biblioteca Lodash

  
    
     Example 1: Using Lodash's _.intersection function

      var animaisArray1 = ['cabra', 'cobra', 'elefante', 'peixe', 'hamster']
      var animaisArray2 = ['cabra', 'elefante', 'pato', 'cavalo', 'gato']

Exemplo de Uso do Spread Operator e Push

  
    
     First Example: Joining elements using spread instead of push

            let primeiros = [1, 2, 3]
            let numeros = [...primeiros, 4,5,6]
            
            console.log(numeros) Saida: [1,2,3,4,5,6]
      

     Second Example: Using the Array Push method 

        let contagem = [4, 5, 6]
        let adicionar = [1,2,3]
        adicionar.push(contagem)

        console.log(adicionar) Saida: [1,2,3,[4,5,6]]

    oque percebemos é que ao usar push ou unshift com dois ou + arrays
    o resultado não será satisfatório, sendo assim melhor usar o spread
